247  @field_validator("hashtags", mode="after")
248  @classmethod
249  def validate_hashtags(cls, hashtags: Optional[str]) -> List[str]:
250  """Validate hashtags.
251 
252  - Receives a string and parsed as a list of tags.
253  - Commas or semicolons are replaced with spaces before splitting.
254  - Add '#' to hashtag if missing.
255  - Also add default '#FMTM' tag.
256  """
257  if hashtags is None:
258  return ["#FMTM"]
259 
260  hashtags = hashtags.replace(",", " ").replace(";", " ")
261  hashtags_list = hashtags.split()
262 
263  # Add '#' to hashtag strings if missing
264  hashtags_with_hash = [
265  f"#{hashtag}" if hashtag and not hashtag.startswith("#") else hashtag
266  for hashtag in hashtags_list
267  ]
268 
269  if "#FMTM" not in hashtags_with_hash:
270  hashtags_with_hash.append("#FMTM")
271 
272  return hashtags_with_hash
