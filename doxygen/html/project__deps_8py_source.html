36 async def get_project_by_id(
37  db: Session = Depends(get_db), project_id: Optional[int] = None
38 ):
39  """Get a single project by id."""
40  if not project_id:
41  # Skip if no project id passed
42  # FIXME why do we need this?
43  return None
44 
45  # TODO replace outline / centroid with geojson equivalent
46  # TODO when we remove SQLALchemy
47  # ST_AsGeoJSON(p.outline)::jsonb AS outline,
48  # ST_AsGeoJSON(p.centroid)::jsonb AS centroid,
49  query = text("""
50  WITH latest_task_history AS (
51  SELECT DISTINCT ON (task_id)
52  task_id,
53  action,
54  action_date,
55  user_id
56  FROM
57  task_history
58  WHERE
59  action != 'COMMENT'
60  ORDER BY
61  task_id, action_date DESC
62  )
63  SELECT
64  p.*,
65  ST_AsGeoJSON(p.outline)::jsonb AS outline,
66  ST_AsGeoJSON(p.centroid)::jsonb AS centroid,
67  JSON_BUILD_OBJECT(
68  'project_id', pi.project_id,
69  'project_id_str', pi.project_id_str,
70  'name', pi.name,
71  'short_description', pi.short_description,
72  'description', pi.description,
73  'text_searchable', pi.text_searchable,
74  'per_task_instructions', pi.per_task_instructions
75  ) AS project_info,
76  JSON_BUILD_OBJECT(
77  'id', project_author.id,
78  'username', project_author.username
79  ) AS author,
80  COALESCE(
81  JSON_AGG(
82  JSON_BUILD_OBJECT(
83  'id', t.id,
84  'project_id', t.project_id,
85  'project_task_index', t.project_task_index,
86  'outline', ST_AsGeoJSON(t.outline)::jsonb,
87  'feature_count', t.feature_count,
88  'task_status', COALESCE(
89  latest_th.action,
90  'RELEASED_FOR_MAPPING'
91  ),
92  'locked_by_uid', COALESCE(
93  latest_user.id,
94  NULL
95  ),
96  'locked_by_username', COALESCE(
97  latest_user.username,
98  NULL
99  )
100  )
101  ) FILTER (WHERE t.id IS NOT NULL), '[]'::json
102  ) AS tasks
103  FROM
104  projects p
105  LEFT JOIN
106  project_info pi ON p.id = pi.project_id
107  LEFT JOIN
108  users project_author ON p.author_id = project_author.id
109  LEFT JOIN
110  tasks t ON p.id = t.project_id
111  LEFT JOIN
112  latest_task_history latest_th ON t.id = latest_th.task_id
113  LEFT JOIN
114  users latest_user ON latest_th.user_id = latest_user.id
115  WHERE
116  p.id = :project_id
117  GROUP BY
118  p.id, pi.project_id, project_author.id;
119  """)
120 
121  result = db.execute(query, {"project_id": project_id})
122  row = result.fetchone()
123 
124  # db_project = db.query(DbProject).filter(DbProject.id == project_id).first()
125  if not row:
126  raise HTTPException(
127  status_code=HTTPStatus.NOT_FOUND,
128  detail=f"Project with ID {project_id} does not exist",
129  )
130 
131  if row.odk_token is None:
132  log.warning(
133  f"Project ({row.id}) has no 'odk_token' set. The QRCode will not work!"
134  )
135 
136  # FIXME Workaround to convert back to SQLAlchemy model for now
137  # FIXME remove this once we remove SQLAlchemy
138  project_dict = {**row._mapping}
139 
140  # FIXME this avoids an error as the fields do not exist on the model
141  task_status_dict = {}
142  task_lock_uid_dict = {}
143  task_lock_user_dict = {}
144  for task in project_dict["tasks"]:
145  task_status_dict[task["id"]] = task["task_status"]
146  del task["task_status"]
147  task_lock_uid_dict[task["id"]] = task["locked_by_uid"]
148  del task["locked_by_uid"]
149  task_lock_user_dict[task["id"]] = task["locked_by_username"]
150  del task["locked_by_username"]
151 
152  db_project = db_models.DbProject(**project_dict)
153 
154  for task in db_project.tasks:
155  task.task_status = task_status_dict[task.id]
156  task.locked_by_uid = task_lock_uid_dict[task.id]
157  task.locked_by_username = task_lock_user_dict[task.id]
158 
159  return db_project

162 @alru_cache(maxsize=32)
163 async def get_odk_credentials(db: Session, project_id: int):
164  """Get ODK credentials of a project, or default organization credentials."""
165  sql = text("""
166  SELECT
167  COALESCE(
168  NULLIF(projects.odk_central_url, ''),
169  organisations.odk_central_url)
170  AS odk_central_url,
171  COALESCE(
172  NULLIF(projects.odk_central_user, ''),
173  organisations.odk_central_user)
174  AS odk_central_user,
175  COALESCE(
176  NULLIF(projects.odk_central_password, ''),
177  organisations.odk_central_password
178  ) AS odk_central_password
179  FROM
180  projects
181  LEFT JOIN
182  organisations ON projects.organisation_id = organisations.id
183  WHERE
184  projects.id = :project_id
185  """)
186  result = db.execute(sql, {"project_id": project_id})
187  creds = result.first()
188 
189  url = creds.odk_central_url
190  user = creds.odk_central_user
191  password = creds.odk_central_password
192 
193  log.debug(f"Retrieved ODK creds for project ({project_id}): {url} | {user}")
194 
195  return project_schemas.ODKCentralDecrypted(
196  odk_central_url=url,
197  odk_central_user=user,
198  odk_central_password=password,
199  )
