294  def __init__(self, **kw):
295  """NOTE this is a workaround for raw SQL queries.
296 
297  The raw SQL .execute returns a Row.
298  We must manually define the relationship models.
299  """
300  mapper = inspect(self).mapper
301 
302  for relationship_key in mapper.relationships.keys():
303  if relationship_key in kw:
304  related_class = mapper.relationships[relationship_key].entity.class_
305  if isinstance(kw[relationship_key], dict):
306  kw[relationship_key] = related_class(**kw[relationship_key])
307  elif isinstance(kw[relationship_key], list):
308  kw[relationship_key] = [
309  related_class(**item) if isinstance(item, dict) else item
310  for item in kw[relationship_key]
311  ]
312 
313  super().__init__(**kw)
